name: Build Windows desktop

on:
  workflow_dispatch:

jobs:
  build-windows:
    runs-on: windows-latest
    steps:
      - uses: actions/checkout@v4
      - uses: subosito/flutter-action@v2
        with:
          channel: stable
          cache: true
      - name: Create Windows host if missing
        shell: pwsh
        run: |
          if (-not (Test-Path windows)) {
            flutter create --platforms=windows .
          }
      - run: flutter pub get
      - run: dart run build_runner build --delete-conflicting-outputs
      - run: flutter analyze
      - run: flutter test
      - run: flutter build windows --release
      - name: Package portable Windows app
        shell: pwsh
        run: Compress-Archive -Path build/windows/x64/runner/Release/* -DestinationPath chronicle-windows-x64.zip
      - uses: actions/upload-artifact@v4
        with:
          name: chronicle-windows-x64
          path: chronicle-windows-x64.zip
